## 十、Worldbuilding 模块架构

Worldbuilding（可视化世界观编辑器）模块提供人物关系图谱、势力分布图、时间线管理三大功能，帮助作者系统性构建和管理故事世界观。

### 10.1 模块定位

- **模块路径**：src/features/worldbuilding/
- **功能定位**：Creator Studio 的独立功能模块，支持作为独立 Tab 或 Modal 打开
- **核心能力**：
  - 人物关系图谱：可视化人物关系网络
  - 势力分布图：管理故事中的组织/势力结构
  - 时间线管理：记录和管理故事事件时间线

    const buildTree = (parentId: string | null): FactionTreeNode[] => {
      return factions
        .filter((f) => f.parentId === parentId)
        .map((f) => ({
          faction: f,
          children: buildTree(f.id),
        }));
    };
    return buildTree(null);
  },
}));

#### 10.4.3 时间线状态 (store/timelineStore.ts)

`	ypescript
import { create } from "zustand";
import type { TimelineEvent, TimelineState } from "../types/timeline";

interface TimelineStore extends TimelineState {
  // Actions
  loadEvents: (projectPath: string) => Promise<void>;
  addEvent: (event: Omit<TimelineEvent, "id" | "createdAt" | "updatedAt">) => Promise<TimelineEvent>;
  updateEvent: (id: string, updates: Partial<TimelineEvent>) => Promise<void>;
  deleteEvent: (id: string) => Promise<void>;

  // 选择和视图
  selectEvent: (id: string | null) => void;
  setCurrentEra: (era: string | undefined) => void;
  setViewMode: (mode: "linear" | "branching") => void;

  // 排序
  reorderEvents: (eventIds: string[]) => void;
}

export const useTimelineStore = create<TimelineStore>((set, get) => ({
  events: [],
  selectedEventId: null,
  currentEra: undefined,
  viewMode: "linear",

  loadEvents: async (projectPath) => {
    // TODO: 从项目目录加载时间线数据
  },

  addEvent: async (data) => {
    const { events } = get();
    const maxOrder = events.reduce((max, e) => Math.max(max, e.order), -1);
    const event: TimelineEvent = {
      ...data,
      id: event_,
      order: data.order ?? maxOrder + 1,
      createdAt: Date.now(),
      updatedAt: Date.now(),
    };
    set((state) => ({ events: [...state.events, event] }));
    return event;
  },

  updateEvent: async (id, updates) => {
    set((state) => ({
      events: state.events.map((e) =>
        e.id === id ? { ...e, ...updates, updatedAt: Date.now() } : e
      ),
    }));
  },

  deleteEvent: async (id) => {
    set((state) => ({
      events: state.events.filter((e) => e.id !== id),
    }));
  },

  selectEvent: (id) => set({ selectedEventId: id }),
  setCurrentEra: (era) => set({ currentEra: era }),
  setViewMode: (mode) => set({ viewMode: mode }),

  reorderEvents: (eventIds) => {
    set((state) => ({
      events: eventIds.map((id, index) => {
        const event = state.events.find((e) => e.id === id);
        return event ? { ...event, order: index } : null;
      }).filter(Boolean) as TimelineEvent[],
    }));
  },
}));
`

// components/Timeline/index.tsx
import { useTimelineStore } from "../../store/timelineStore";
import { EventCard } from "./EventCard";
import { Empty, Segmented } from "antd";

export function Timeline() {
  const { events, viewMode, setViewMode, currentEra } = useTimelineStore();

  // 按时间排序
  const sortedEvents = [...events].sort((a, b) => a.order - b.order);

  // 按时代分组（如果设置了时代）
  const groupedEvents = currentEra
    ? { [currentEra]: sortedEvents }
    : sortedEvents.reduce<Record<string, typeof sortedEvents>>((acc, e) => {
        const era = e.era || "默认";
        (acc[era] = acc[era] || []).push(e);
        return acc;
      }, {});

  return (
    <div style={{ padding: 16 }}>
      <div style={{ marginBottom: 16, display: "flex", justifyContent: "space-between" }}>
        <Segmented
          options={[
            { label: "线性", value: "linear" },
            { label: "分支", value: "branching" },
          ]}
          value={viewMode}
          onChange={(value) => setViewMode(value as "linear" | "branching")}
        />
      </div>

      {sortedEvents.length === 0 ? (
        <Empty description="暂无事件，点击添加" />
      ) : (
        <div style={{ display: "flex", flexDirection: "column", gap: 16 }}>
          {sortedEvents.map((event) => (
            <EventCard key={event.id} event={event} />
          ))}
        </div>
      )}
    </div>
  );
}
