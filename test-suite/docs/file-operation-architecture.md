### 7.4 自动化测试脚本示例

```typescript
// test-suite/cases/file-operation.test.ts

describe('保存状态管理测试', () => {
  // TC-SAVE-001: 打开项目后保存状态应为已保存
  it('打开项目后保存状态应为已保存', async () => {
    // 1. 打开项目
    await openProject();
    
    // 2. 选择章节
    await selectChapter('chapter-1');
    
    // 3. 验证状态显示为已保存
    const statusText = await getSaveStatusText();
    expect(statusText).toBe('已保存');
  });

  // TC-SAVE-002: 修改内容后状态变为未保存
  it('修改内容后状态变为未保存', async () => {
    // 1. 选择章节
    await selectChapter('chapter-1');
    
    // 2. 修改内容
    await editor.type('测试内容');
    
    // 3. 验证状态变为未保存
    const statusText = await getSaveStatusText();
    expect(statusText).toBe('未保存');
  });

  // TC-SAVE-006: 内容未实际变化不触发未保存（BUG-001回归验证）
  it('内容未实际变化不触发未保存状态', async () => {
    // 1. 打开项目选择章节
    await selectChapter('chapter-1');
    
    // 2. 等待初始加载完成
    await waitForSaveStatus('已保存');
    
    // 3. 执行不改变内容的操作
    await editor.moveCursor(0);
    await editor.pressKey('End');
    await editor.pressKey('Home');
    
    // 4. 验证状态保持已保存
    const statusText = await getSaveStatusText();
    expect(statusText).toBe('已保存');
  });
});

describe('导出功能测试', () => {
  // TC-EXPORT-001: 导出单个章节
  it('应该能正确导出单个章节为TXT', async () => {
    // 1. 选择章节
    await selectChapter('chapter-1');
    
    // 2. 打开导出对话框
    await clickExportButton();
    
    // 3. 选择导出当前章节
    await selectExportMode('single');
    
    // 4. 选择保存位置
    await selectSaveLocation('/tmp/test-export');
    
    // 5. 确认导出
    await confirmExport();
    
    // 6. 验证导出成功
    await expectExportSuccess();
    expect(await fileExists('/tmp/test-export/章节.txt')).toBe(true);
  });
});
```

### 7.6 测试风险评估

| 风险项 | 影响程度 | 发生可能性 | 风险等级 | 应对措施 |
|-------|---------|-----------|---------|---------|
| 快速连续操作导致状态不同步 | 高 | 低 | 中 | 增加操作间隔，或添加防抖机制 |
| 自动保存与手动保存冲突 | 中 | 中 | 中 | 确保自动保存有优先级机制 |
| 网络断开导致保存失败 | 高 | 低 | 低 | 添加网络状态检测和重试机制 |
| 超大文件导致性能问题 | 中 | 低 | 低 | 添加文件大小限制和性能监控 |

### 7.7 后续测试建议
